class Solution {
public:
    int snowball(int N, vector<int> weights){
        int sum=0;
        for(int i=0;i<weights.size();i++){
            sum+=weights.at(i);
        }
        return sum;
    }
};

    int T;
    cin >> T;
    while(T--){
        int N;
        cin >> N;
        vector<int> weights(N);
        for(int i = 0; i < N; i++){
            cin >> weights[i];
        }
        Solution obj;
        cout << obj.snowball(N, weights) << "\n";
    }
}
